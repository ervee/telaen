        function pout($fp, $value)
        {
            switch (gettype($value)) {
                case 'boolean':
                    fwrite($fp, ($value ? "true" : "false"));
                    break;
                case 'integer':
                case 'double':
                    fwrite($fp, $value);
                    break;
                case 'string':
                    $value = preg_replace("|'|", "\'", $value);
                    fwrite($fp, "'$value'");
                    break;
                case 'array':
                    fwrite($fp, "array(\n");
                    foreach ($value as $a => $b) {
                        fwrite($fp, "   ");
                        pout($fp, $a);
                        fwrite($fp, " => ");
                        pout($fp, $b);
                        fwrite($fp, ",\n");
                    }
                    fwrite($fp, ")");
                    break;
                default:
                    fwrite($fp, "'$value'"); break;
            }
        }
